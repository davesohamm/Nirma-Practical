\section{Objective}
This lab explores advanced interpretable models and visualization techniques for explainability:
\begin{enumerate}
    \item \textbf{Generalized Linear Models (GLM)} --- extending linear regression with non-Gaussian distributions
    \item \textbf{Generalized Additive Models (GAM)} --- capturing non-linear feature effects with splines
    \item \textbf{Partial Dependence Plots (PDP)} --- visualizing marginal feature effects
    \item \textbf{SHAP Values} --- computing Shapley-based feature attributions for GAMs
    \item \textbf{Individual Conditional Expectation (ICE) Plots} --- revealing heterogeneous feature effects
\end{enumerate}

\section{Methodology}

\subsection{Generalized Linear Models (GLM)}

\subsubsection{Theory}
A GLM extends ordinary linear regression by allowing the response variable to follow any distribution from the exponential family. A GLM consists of three components:
\begin{itemize}
    \item \textbf{Random component:} The distribution of $Y$ (e.g., Gaussian, Gamma, Binomial)
    \item \textbf{Systematic component:} The linear predictor $\eta = \mathbf{X}\boldsymbol{\beta}$
    \item \textbf{Link function:} $g(\mu) = \eta$, connecting the mean response to the linear predictor
\end{itemize}

\begin{equation}
    g(E[Y \mid \mathbf{X}]) = \beta_0 + \beta_1 x_1 + \cdots + \beta_p x_p
\end{equation}

\subsubsection{GLM for Diabetes (Regression)}
The diabetes target is continuous, positive, and right-skewed. Therefore:
\begin{itemize}
    \item \textbf{Family:} Gamma (appropriate for positive, skewed data)
    \item \textbf{Link:} Log (ensures predictions remain positive)
\end{itemize}

With the log link, coefficients have a \textbf{multiplicative interpretation}: $e^{\beta_j}$ gives the multiplicative change in the expected response for a unit increase in $x_j$.

\subsubsection{GLM for Breast Cancer (Classification)}
For binary classification, the appropriate choice is:
\begin{itemize}
    \item \textbf{Family:} Binomial
    \item \textbf{Link:} Logit (maps probabilities to $(-\infty, +\infty)$)
\end{itemize}

\begin{equation}
    \text{logit}(P(Y=1)) = \log\left(\frac{P(Y=1)}{1 - P(Y=1)}\right) = \beta_0 + \sum_{j=1}^{p} \beta_j x_j
\end{equation}

\subsection{Generalized Additive Models (GAM)}

\subsubsection{Theory}
GAMs extend GLMs by replacing linear terms with smooth, non-linear functions:
\begin{equation}
    g(E[Y]) = \beta_0 + f_1(x_1) + f_2(x_2) + \cdots + f_p(x_p)
\end{equation}

Each $f_j$ is typically a spline function that can capture non-linear relationships. The \textbf{Effective Degrees of Freedom (EDoF)} of each spline indicates the complexity of the learned shape:
\begin{itemize}
    \item EDoF $\approx 1$: Nearly linear relationship
    \item EDoF $> 3$: Complex, non-linear relationship
\end{itemize}

\subsubsection{Implementation}
\begin{itemize}
    \item \textbf{Diabetes:} \texttt{LinearGAM} with 10 spline terms, optimized via \texttt{gridsearch()}
    \item \textbf{Results:} MSE = 2750.61, $R^2 = 0.48$
    \item \textbf{Cancer:} \texttt{LogisticGAM} with 10 spline terms
    \item \textbf{Results:} Accuracy = 96.49\%, Confusion Matrix: $[[41, 2], [2, 69]]$
\end{itemize}

\subsection{Partial Dependence Plots (PDP)}

\subsubsection{Theory}
PDPs show the marginal effect of a feature $x_j$ on the predicted outcome, averaging over all other features:
\begin{equation}
    \hat{f}_j(x_j) = \frac{1}{n} \sum_{i=1}^{n} f(x_j, \mathbf{x}_{i,-j})
\end{equation}

This reveals the ``average'' relationship between a feature and the prediction.

\subsection{SHAP Analysis on GAM}

\subsubsection{Theory}
SHAP values decompose a prediction into additive feature contributions based on cooperative game theory:
\begin{equation}
    \phi_j = \sum_{S \subseteq N \setminus \{j\}} \frac{|S|!(|N|-|S|-1)!}{|N|!} \left[ f(S \cup \{j\}) - f(S) \right]
\end{equation}

Since GAMs are not tree-based, \texttt{KernelExplainer} from SHAP is used, which works with any model by perturbing inputs and observing predictions.

\subsection{Individual Conditional Expectation (ICE) Plots}

\subsubsection{Theory}
ICE plots extend PDPs by showing the feature-response curve for \textbf{each individual observation}, rather than just the average:
\begin{equation}
    \hat{f}_j^{(i)}(x_j) = f(x_j, \mathbf{x}_{i,-j})
\end{equation}

\textbf{Centered ICE (c-ICE) plots} subtract the value at the minimum point to highlight differences in slopes:
\begin{equation}
    \hat{f}_{j,\text{centered}}^{(i)}(x_j) = \hat{f}_j^{(i)}(x_j) - \hat{f}_j^{(i)}(x_{j,\min})
\end{equation}

\section{Results and Discussion}
\begin{itemize}
    \item \textbf{GLM:} The Gamma GLM with log link identified \texttt{bmi} and \texttt{s5} as the strongest predictors of diabetes progression. The Binomial GLM correctly identified \texttt{mean radius} as the most critical feature for malignancy.
    \item \textbf{GAM:} By allowing non-linear feature shapes, GAMs captured complex relationships that linear models miss. The cancer GAM achieved 96.49\% accuracy with only 4 misclassifications.
    \item \textbf{PDP:} Partial dependence plots revealed that features like \texttt{bmi} have a non-linear, accelerating effect on diabetes progression --- information lost in linear coefficient summaries.
    \item \textbf{SHAP:} Confirmed that \texttt{bmi} and \texttt{s5} dominate predictions, while providing individual-level explanations through force plots.
    \item \textbf{ICE:} Revealed that patients respond differently to the same feature changes --- some show steep increases in risk with higher \texttt{bmi}, while others are relatively unaffected. This heterogeneity is hidden in averaged PDPs.
\end{itemize}

% -------------------------------------------------------
\section{Conclusion}
This lab demonstrated a progression of explainability techniques from simple coefficient analysis (GLM) to rich, individualized explanations (ICE plots and SHAP). Key takeaways:
\begin{itemize}
    \item GLMs provide a principled extension of linear regression with interpretable coefficients
    \item GAMs capture non-linear effects while maintaining additive interpretability
    \item PDPs summarize the average effect of each feature, while ICE plots reveal individual variation
    \item SHAP values provide a theoretically rigorous framework for both global and local explanations
\end{itemize}
