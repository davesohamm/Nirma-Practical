\section{Objective}
This lab explores three gradient-based methods for explaining predictions of deep neural networks in computer vision:
\begin{enumerate}
    \item \textbf{Saliency Maps} --- identifying influential pixels via input gradients
    \item \textbf{Integrated Gradients} --- attributing predictions through path integration
    \item \textbf{Grad-CAM} --- generating class-discriminative localization maps from convolutional layers
\end{enumerate}

All three methods are applied to a pre-trained \textbf{ResNet-18} model on the \textbf{Flowers102} dataset.

% -------------------------------------------------------
\section{Dataset and Model}

\subsection{Flowers102 Dataset}
\begin{itemize}
    \item \textbf{Source:} Oxford 102 Flower Category Dataset (\texttt{torchvision.datasets.Flowers102})
    \item \textbf{Task:} Image classification (102 flower categories)
    \item \textbf{Test Image:} Sample at index 11 from the test split
    \item \textbf{Input Resolution:} $448 \times 448$ pixels
    \item \textbf{Preprocessing:} Resize $\rightarrow$ ToTensor $\rightarrow$ ImageNet normalization (mean = $[0.485, 0.456, 0.406]$, std = $[0.229, 0.224, 0.225]$)
\end{itemize}

\subsection{Model: ResNet-18}
\begin{itemize}
    \item \textbf{Architecture:} 18-layer Residual Network with skip connections
    \item \textbf{Pre-trained:} ImageNet weights (1000 classes)
    \item \textbf{Key layers:} conv1 $\rightarrow$ layer1 $\rightarrow$ layer2 $\rightarrow$ layer3 $\rightarrow$ layer4 $\rightarrow$ avgpool $\rightarrow$ fc
    \item \textbf{Target layer for Grad-CAM:} \texttt{layer4} (last convolutional block)
\end{itemize}

\section{Methodology}

\subsection{Saliency Maps}

\subsubsection{Theory}
A saliency map computes the gradient of the output score with respect to the input image:
\begin{equation}
    S_{ij} = \left| \frac{\partial y_c}{\partial x_{ij}} \right|
\end{equation}

where $y_c$ is the score for the predicted class $c$, and $x_{ij}$ is the pixel at position $(i,j)$. The absolute gradient magnitude indicates how sensitive the prediction is to small changes in each pixel.

For a color image with $C$ channels, the saliency at each spatial location is:
\begin{equation}
    S_{ij} = \max_{c \in \{R,G,B\}} \left| \frac{\partial y_c}{\partial x_{c,ij}} \right|
\end{equation}

\subsubsection{Implementation Steps}
\begin{enumerate}
    \item Set \texttt{input\_tensor.requires\_grad = True} to enable gradient computation
    \item Performed a forward pass to obtain the predicted class score
    \item Backpropagated the score to compute gradients w.r.t.\ the input
    \item Took the absolute value and maximum across color channels
    \item Applied spatial smoothing with \texttt{F.avg\_pool2d} (kernel size = 7) to reduce noise
    \item Overlaid the saliency map on the original image using jet colormap with $\alpha = 0.4$ transparency
\end{enumerate}

\subsection{Integrated Gradients}

\subsubsection{Theory}
Integrated Gradients (IG) compute attributions by integrating gradients along a path from a baseline $\mathbf{x}'$ (typically a black image) to the actual input $\mathbf{x}$:

\begin{equation}
    \text{IG}_j(\mathbf{x}) = (x_j - x'_j) \times \int_{\alpha=0}^{1} \frac{\partial f(\mathbf{x}' + \alpha(\mathbf{x} - \mathbf{x}'))}{\partial x_j} \, d\alpha
\end{equation}

In practice, the integral is approximated with a Riemann sum over $m$ steps:
\begin{equation}
    \text{IG}_j(\mathbf{x}) \approx (x_j - x'_j) \times \frac{1}{m} \sum_{k=1}^{m} \frac{\partial f(\mathbf{x}' + \frac{k}{m}(\mathbf{x} - \mathbf{x}'))}{\partial x_j}
\end{equation}

\subsubsection{Key Properties}
IG satisfies two important axioms:
\begin{itemize}
    \item \textbf{Sensitivity:} If the input and baseline differ in one feature and produce different predictions, that feature receives a non-zero attribution
    \item \textbf{Completeness:} The sum of all attributions equals the difference between $f(\mathbf{x})$ and $f(\mathbf{x}')$:
    \begin{equation}
        \sum_j \text{IG}_j(\mathbf{x}) = f(\mathbf{x}) - f(\mathbf{x}')
    \end{equation}
\end{itemize}

\subsubsection{Implementation}
\begin{enumerate}
    \item Set baseline to a black image (all zeros)
    \item Used $m = 100$ interpolation steps
    \item At each step: computed the interpolated input, performed a forward pass, computed gradients for the target class
    \item Averaged gradients across all steps and multiplied by $(x - x')$
    \item Summed absolute attributions across color channels and normalized to $[0, 1]$
\end{enumerate}

\begin{figure}[H]
    \centering
    \includegraphics[width=\textwidth]{images_5/output_2.png}
    \caption{Left: Original flower image. Right: Integrated Gradients attribution map. Compared to the saliency map, IG provides more distributed and robust highlighting of the flower region, satisfying the completeness axiom.}
    \label{fig:ig}
\end{figure}

\subsection{Grad-CAM}

\subsubsection{Theory}
Grad-CAM (Gradient-weighted Class Activation Mapping) generates coarse localization maps by leveraging the spatial information in convolutional feature maps. For a target class $c$ and feature maps $A^k$ from the target layer:

\begin{equation}
    \alpha_k^c = \frac{1}{Z} \sum_i \sum_j \frac{\partial y^c}{\partial A_{ij}^k} \quad \text{(Global Average Pooling of gradients)}
\end{equation}

\begin{equation}
    L_{\text{Grad-CAM}}^c = \text{ReLU}\left( \sum_k \alpha_k^c A^k \right)
\end{equation}

The ReLU ensures we only consider features that have a \textbf{positive} influence on the target class.

\subsubsection{Implementation}
\begin{enumerate}
    \item Implemented a \texttt{GradCAM} class with:
    \begin{itemize}
        \item Forward hooks on \texttt{layer4} to capture activations $A^k$
        \item Backward hooks to capture gradients $\frac{\partial y^c}{\partial A^k}$
    \end{itemize}
    \item Computed importance weights $\alpha_k^c$ via global average pooling of gradients
    \item Generated the weighted sum of activation maps followed by ReLU
    \item Resized the $14 \times 14$ Grad-CAM heatmap to $448 \times 448$ using bilinear interpolation
    \item Overlaid the heatmap on the original image with transparency
\end{enumerate}

\end{figure}

\subsection{Discussion}
\begin{itemize}
    \item \textbf{Saliency maps} are the simplest and fastest method but suffer from noise and gradient saturation. Some important pixels may receive zero saliency if the model is locally flat at that point.
    
    \item \textbf{Integrated Gradients} addresses these limitations by accumulating gradients along a path from a baseline. The result is more robust, though computationally more expensive ($100\times$ more forward passes in our implementation).
    
    \item \textbf{Grad-CAM} operates at the level of convolutional feature maps rather than individual pixels, producing semantically meaningful but coarser explanations. It is class-discriminative --- the heatmap changes when a different target class is selected.
    
    \item All three methods agree that the \textbf{flower region} (petals and center) is the primary focus of the model's prediction, while the background is largely ignored. This builds confidence that the model has learned meaningful visual features rather than spurious correlations.
\end{itemize}

\section{Conclusion}
This lab demonstrated three complementary approaches to explaining deep neural network predictions in computer vision:
\begin{itemize}
    \item \textbf{Saliency maps} provide a quick, pixel-level view of model sensitivity
    \item \textbf{Integrated Gradients} offer theoretically grounded attributions that satisfy key axioms
    \item \textbf{Grad-CAM} provides intuitive, region-level explanations by leveraging convolutional feature maps
\end{itemize}

Together, these methods form a comprehensive toolkit for understanding what deep learning models ``see'' and ``focus on'' when making predictions. This is crucial for building trust in AI systems deployed in safety-critical applications such as medical imaging and autonomous driving.
